# Automated commit and push
clg() {
    unset ANTHROPIC_API_KEY
    claude -p -c "Your task is to commit and push
ONLY the uncommitted changes in this git repository.

WORKFLOW:
1. Run 'git status' to see uncommitted files
2. Run 'git diff' to review changes
3. Stage changes with 'git add'
4. Commit with a descriptive message
5. Push with 'git push'

DO NOT reference already-committed changes.
After completing, output a brief success message
and exit immediately." \
    --allowedTools "Bash(git status:*)" \
                   "Bash(git diff:*)" \
                   "Bash(git add:*)" \
                   "Bash(git commit:*)" \
                   "Bash(git push:*)"
}
